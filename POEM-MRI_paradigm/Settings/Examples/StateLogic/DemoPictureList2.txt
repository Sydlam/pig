//	Description:
//		Sets up a PictureList so that it can be used.
//		Sets up fkey 12 to sequence through the pictureList and then stop.
//		Sets up fkey 11 to reset the picture list.
//
//	The image file names are by default relative to the folder: ViewPoint/Images/
//	unless a setPath cmd has changed this.
//	For example, "Misc/045.bmp" specifies: "ViewPoint/Images/Misc/045.bmp"
//
pictureList_Init
pictureList_AddName "Misc/ViewPointMan.bmp"
pictureList_AddName "Misc/TwoMenOnStage.bmp"
pictureList_AddName "Misc/CARNAC.bmp"
pictureList_AddName "Misc/ARI_flyer.bmp"
pictureList_AddName "Misc/ARI_Logo.bmp"
pictureList_AddName "Misc/Gabor.bmp"
pictureList_AddName "Misc/apple.bmp"
pictureList_AddName "Misc/BlankGrayImage.bmp"  // pad at end, if using pictureList_EndAction
//
//
fkey_cmd  6 { pictureList_Restart }     // removes "finished" flag and sets index to #1
fkey_cmd  7 { if {PictureListFinishedQ}{say"EndOfList"} {pictureList_ShowNext} }
fkey_cmd  8 { pictureList_ShowIndex 3 } // sets index to #3, shows #3, then increments index to #4
fkey_cmd  9 { pictureList_Randomize Off }
fkey_cmd 10 { pictureList_Randomize }   // optional parameter for backward compatibility
//
//	ttl_cmd +0  { pictureList_ShowNext }
//	ttl_cmd -0  { stimulus_loadImageFile  "Misc/BlankGrayImage.BMP" }
END
